void HAL_TIM_IC_CaptureCallback(TIM_HandleTypeDef *htim)
{
	if ((htim == &htim1) && (htim->Channel == 1))
	{
		if (flag == 0)
		{
			count_1 = HAL_TIM_ReadCapturedValue(&htim1, TIM_CHANNEL_1);
			flag = 1;
		}
		else
		{
			count_2 = HAL_TIM_ReadCapturedValue(&htim1, TIM_CHANNEL_1);
			__HAL_TIM_SET_COUNTER(htim, 0);
			distance = (count_2 - count_1)/58.0;
			flag = 0;
			if (distance>distanceinRange)
			{
				distanceFlag=0;
			}
			else if (distance< distanceinRange)
			{
				distanceFlag=1;
			}
		}
	}
}

void HAL_UART_RxCpltCallback(UART_HandleTypeDef *huart)
{
	if (huart == &huart2)
	{
		if (flagFromPython == '1')
		{
			manual_mode_flag = 1;
	        distance_mode_flag = 0;

		}
		else if (flagFromPython =='2')
		{
            manual_mode_flag = 0;
            distance_mode_flag = 1;
		}
	}
	HAL_UART_Receive_IT(&huart2, (uint8_t*)flagFromPython, 1);
}





///////
void HAL_SPI_RxCpltCallback(SPI_HandleTypeDef * hspi)
{
	// SLAVE
	if (hspi == &hspi1)
	{
		//only open spi transmission when its in normal operation mode or when trigger mode is on AND distance is within range
		if(manual_mode_flag || (distance_mode_flag && distanceFlag))
		{
			LSB_8 = RX_Buffer[0]; // extract the least significant 8 bits of the ADC value
			MSB_8 = RX_Buffer[1]; // extract the most significant 8 bits of the ADC value

			// Construct 16-bit unsigned ADC value
			ADC_output = MSB_8;
			ADC_output = ADC_output << 8;
			ADC_output = ADC_output | LSB_8;

			// filtering
			// moving average filter
			// store in ADC_output_array
			ADC_output_array[average_index] = ADC_output;

			if (average_counter > 2)
			{
				moving_average = (ADC_output_array[0] + ADC_output_array[1] + ADC_output_array[2]) / 3;
				// Transmit string_1 over UART under interrupt mode
				HAL_UART_Transmit_IT(&huart2, (uint8_t*)RX_Buffer, 2);
			}
			else
			{
				// Transmit string_1 over UART under interrupt mode
				HAL_UART_Transmit_IT(&huart2, (uint8_t*)RX_Buffer, 2);
			}

			average_counter++;
			// average index has to go back to 0 after 2, making FIFO
			average_index = (average_counter - 1) % 3;

			HAL_SPI_Receive_IT(&hspi1, RX_Buffer, 2); // Receive through SPI in interrupt mode

		}
	}
}
